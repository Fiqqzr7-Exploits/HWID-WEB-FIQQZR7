        let keys = [];
        
        function generateKey() {
            const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789';
            const part1 = Array(4).fill().map(() => 
                chars[Math.floor(Math.random() * chars.length)]).join('');
            const part2 = Array(4).fill().map(() => 
                chars[Math.floor(Math.random() * chars.length)]).join('');
            return `FIQ-${part1}-${part2}`;
        }
        
        function generateKeys() {
            keys = [];
            const total = 200;
            const container = document.getElementById('keyContainer');
            const progressBar = document.getElementById('progressBar');
            
            container.innerHTML = '';
            
            for (let i = 0; i < total; i++) {
                keys.push(generateKey());
                
                // Update progress
                const percent = ((i + 1) / total) * 100;
                progressBar.style.width = percent + '%';
                
                // Add to display (every 10 keys to avoid lag)
                if (i % 10 === 0 || i === total - 1) {
                    updateDisplay();
                }
            }
            
            updateStats();
            progressBar.style.width = '100%';
        }
        
        function updateDisplay() {
            const container = document.getElementById('keyContainer');
            container.innerHTML = '';
            
            keys.forEach((key, index) => {
                const keyDiv = document.createElement('div');
                keyDiv.className = 'key-item';
                keyDiv.innerHTML = `
                    <span>${index + 1}. ${key}</span>
                    <button class="copy-btn" onclick="copyKey('${key}')">COPY</button>
                `;
                container.appendChild(keyDiv);
            });
        }
        
        function updateStats() {
            document.getElementById('totalKeys').textContent = keys.length;
            document.getElementById('availableKeys').textContent = keys.length;
            document.getElementById('usedKeys').textContent = 0;
        }
        
        function copyKey(key) {
            navigator.clipboard.writeText(key).then(() => {
                alert('âœ… Key copied: ' + key);
            });
        }
        
        function copyAllKeys() {
            const text = keys.join('\n');
            navigator.clipboard.writeText(text).then(() => {
                alert('âœ… ' + keys.length + ' keys copied to clipboard!');
            });
        }
        
        function clearKeys() {
            if (confirm('Clear all keys?')) {
                keys = [];
                document.getElementById('keyContainer').innerHTML = 
                    '<div style="text-align: center; color: #8a9bb5; padding: 50px;">No keys generated</div>';
                updateStats();
                document.getElementById('progressBar').style.width = '0%';
            }
        }
